### B. Create `src/app/api/tools/route.ts`

This route merges two data sources:
1. `~/clawd/tools-health.json` — the cached output of the health-check script (cli/plugins/scripts sections)
2. `~/.openclaw/cron/jobs.json` — real-time cron job state (cron section)

```typescript
import { NextResponse } from "next/server";
import { readFile } from "fs/promises";
import type { ToolsHealth, CronEntry, HealthStatus } from "@/lib/types/tools";

export const dynamic = "force-dynamic";
export const runtime = "nodejs";

const TOOLS_HEALTH_PATH = "/home/ubuntu/clawd/tools-health.json";
const JOBS_PATH = "/home/ubuntu/.openclaw/cron/jobs.json";

interface RawCronJob {
  id: string;
  name: string;
  enabled: boolean;
  schedule: { expr: string; tz?: string };
  state?: {
    nextRunAtMs?: number;
    lastRunAtMs?: number;
    lastStatus?: string;
    consecutiveErrors?: number;
  };
}

function getCronHealth(job: RawCronJob): HealthStatus {
  if (!job.enabled) return "red";
  const errors = job.state?.consecutiveErrors ?? 0;
  if (errors >= 2) return "red";
  if (errors === 1) return "yellow";
  if (!job.state?.lastRunAtMs) return "yellow"; // never ran
  return "green";
}

function toCronEntry(job: RawCronJob): CronEntry {
  return {
    id: job.id,
    name: job.name,
    schedule: job.schedule.expr,
    tz: job.schedule.tz ?? "UTC",
    enabled: job.enabled,
    status: getCronHealth(job),
    lastRunAt: job.state?.lastRunAtMs
      ? new Date(job.state.lastRunAtMs).toISOString()
      : null,
    nextRunAt: job.state?.nextRunAtMs
      ? new Date(job.state.nextRunAtMs).toISOString()
      : null,
    lastStatus: job.state?.lastStatus ?? null,
    consecutiveErrors: job.state?.consecutiveErrors ?? 0,
  };
}

export async function GET() {
  try {
    // Read tools-health.json (cli/plugins/scripts)
    const rawHealth = await readFile(TOOLS_HEALTH_PATH, "utf-8");
    const health = JSON.parse(rawHealth) as Omit<ToolsHealth, "cron">;

    // Read jobs.json for cron section (use state.lastRunAtMs, NOT lastRun)
    let cronEntries: CronEntry[] = [];
    try {
      const rawJobs = await readFile(JOBS_PATH, "utf-8");
      const jobs: RawCronJob[] = JSON.parse(rawJobs);
      cronEntries = jobs.map(toCronEntry).sort((a, b) =>
        a.name.localeCompare(b.name)
      );
    } catch {
      // jobs.json unreadable — return empty cron section, not a full error
    }

    const result: ToolsHealth = {
      lastChecked: health.lastChecked ?? null,
      cli: health.cli ?? [],
      plugins: health.plugins ?? [],
      scripts: health.scripts ?? [],
      cron: cronEntries,
    };

    return NextResponse.json(result);
  } catch (error) {
    return NextResponse.json(
      {
        lastChecked: null,
        cli: [],
        plugins: [],
        scripts: [],
        cron: [],
        error: String(error),
      },
      { status: 500 }
    );
  }
}
```

Key implementation notes:
- `state.lastRunAtMs` (milliseconds) is the correct field — NOT `lastRun` (which is often empty per research Pitfall 3)
- `force-dynamic` prevents Next.js from statically caching filesystem reads (Pitfall 6)
- Cron section sorted by name for consistent display
- If jobs.json fails, return empty cron array rather than 500 (graceful degradation)

### A. Create `src/app/api/tools/refresh/route.ts`

The refresh endpoint spawns the health-check script detached (fire-and-forget) and returns 202 immediately. This avoids Next.js API timeout if the script takes more than a few seconds.

```typescript
import { NextResponse } from "next/server";
import { spawn } from "child_process";

export const dynamic = "force-dynamic";
export const runtime = "nodejs";

const SCRIPT_PATH = "/home/ubuntu/scripts/tools-health-check.sh";

export async function POST() {
  try {
    // Fire-and-forget: spawn detached so the API route returns immediately
    const child = spawn(SCRIPT_PATH, [], {
      detached: true,
      stdio: "ignore",
    });
    child.unref();

    return NextResponse.json(
      { ok: true, message: "Health check triggered" },
      { status: 202 }
    );
  } catch (error) {
    return NextResponse.json(
      { ok: false, error: String(error) },
      { status: 500 }
    );
  }
}
```
