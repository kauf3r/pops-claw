**On EC2** via SSH (`ssh -i ~/.ssh/clawdbot-key.pem ubuntu@100.72.143.9`):

Create `/home/ubuntu/scripts/tools-health-check.sh` with the following content. Use `tee` (not heredoc with leading spaces — causes hangs over SSH per project lessons):

```bash
#!/bin/bash
# /home/ubuntu/scripts/tools-health-check.sh
# Runs every 5 min via cron, writes /home/ubuntu/clawd/tools-health.json
# Executed by: */5 * * * * /home/ubuntu/scripts/tools-health-check.sh

set -euo pipefail
export PATH="/home/ubuntu/.npm-global/bin:/usr/local/bin:/usr/bin:/bin"
OUT="/home/ubuntu/clawd/tools-health.json"
LAST_CHECKED=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

# --- CLI Tools ---
OPENCLAW_VER=$(/home/ubuntu/.npm-global/bin/openclaw --version 2>/dev/null | head -1 || echo "error")
GOG_VER=$(/usr/local/bin/gog --version 2>/dev/null | head -1 || echo "error")
NODE_VER=$(/usr/bin/node --version 2>/dev/null || echo "error")
NPM_VER=$(/usr/bin/npm --version 2>/dev/null || echo "error")
# whisper --version prints usage and exits 1 — use pip show instead
WHISPER_VER=$(/home/ubuntu/whisper-venv/bin/pip show openai-whisper 2>/dev/null | grep '^Version' | awk '{print $2}' || echo "error")
# bd is not installed on EC2 (Mac-side tool) — always red
BD_VER="not installed"

# --- Plugins ---
CAMOFOX_DIR=$(test -d /home/ubuntu/.openclaw/extensions/camofox-browser && echo true || echo false)
SECURECLAW_DIR=$(test -d /home/ubuntu/.openclaw/plugins/secureclaw && echo true || echo false)

# --- Scripts ---
PRUNE_MTIME=$(stat -c %Y /home/ubuntu/scripts/prune-sessions.sh 2>/dev/null || echo 0)
VOICENOTES_MTIME=$(stat -c %Y /home/ubuntu/scripts/process-voice-notes.py 2>/dev/null || echo 0)
CLAWDSTRIKE_PATH="/home/ubuntu/.openclaw/skills/clawdstrike/scripts/collect_verified.sh"
CLAWDSTRIKE_MTIME=$(stat -c %Y "$CLAWDSTRIKE_PATH" 2>/dev/null || echo 0)

python3 - <<PYEOF
import json, time, sys

now = time.time()
last_checked = "${LAST_CHECKED}"

def clamp_status(version):
    return "red" if version in ("error", "not installed", "") else "green"

def days_since(mtime_str):
    m = int(mtime_str)
    return round((now - m) / 86400, 1) if m > 0 else None

def script_status(mtime_str):
    m = int(mtime_str)
    return "green" if m > 0 else "red"

data = {
    "lastChecked": last_checked,
    "cli": [
        {
            "id": "openclaw",
            "name": "openclaw",
            "version": "${OPENCLAW_VER}",
            "path": "/home/ubuntu/.npm-global/bin/openclaw",
            "status": clamp_status("${OPENCLAW_VER}"),
            "commands": [
                "openclaw --version",
                "systemctl --user status openclaw-gateway",
                "openclaw doctor"
            ]
        },
        {
            "id": "gog",
            "name": "gog",
            "version": "${GOG_VER}",
            "path": "/usr/local/bin/gog",
            "status": clamp_status("${GOG_VER}"),
            "commands": [
                "gog --version",
                "gog auth list",
                "gog health"
            ]
        },
        {
            "id": "node",
            "name": "node / npm",
            "version": "node ${NODE_VER} / npm ${NPM_VER}",
            "path": "/usr/bin/node",
            "status": clamp_status("${NODE_VER}"),
            "commands": [
                "node --version",
                "npm --version"
            ]
        },
        {
            "id": "whisper",
            "name": "whisper",
            "version": "${WHISPER_VER}",
            "path": "/home/ubuntu/whisper-venv/bin/whisper",
            "status": clamp_status("${WHISPER_VER}"),
            "commands": [
                "source ~/whisper-venv/bin/activate && pip show openai-whisper",
                "ls ~/whisper-venv/lib/"
            ]
        },
        {
            "id": "bd",
            "name": "bd",
            "version": "not installed",
            "path": "N/A",
            "status": "red",
            "commands": []
        }
    ],
    "plugins": [
        {
            "id": "camofox",
            "name": "camofox-browser",
            "version": "1.0.13",
            "status": "green" if ${CAMOFOX_DIR} else "red",
            "dirExists": ${CAMOFOX_DIR},
            "inConfig": True,
            "commands": [
                "openclaw plugins list",
                "ls ~/.openclaw/extensions/camofox-browser/"
            ]
        },
        {
            "id": "secureclaw",
            "name": "secureclaw",
            "version": "2.1.0",
            "status": "green" if ${SECURECLAW_DIR} else "red",
            "dirExists": ${SECURECLAW_DIR},
            "inConfig": True,
            "commands": [
                "ls ~/.openclaw/plugins/secureclaw/",
                "cat ~/.openclaw/plugins/secureclaw/package.json | python3 -m json.tool"
            ]
        }
    ],
    "scripts": [
        {
            "id": "prune-sessions",
            "name": "prune-sessions.sh",
            "path": "/home/ubuntu/scripts/prune-sessions.sh",
            "status": script_status("${PRUNE_MTIME}"),
            "lastModifiedDays": days_since("${PRUNE_MTIME}"),
            "commands": [
                "bash ~/scripts/prune-sessions.sh",
                "tail -20 ~/scripts/prune-sessions.log"
            ]
        },
        {
            "id": "voice-notes",
            "name": "process-voice-notes.py",
            "path": "/home/ubuntu/scripts/process-voice-notes.py",
            "status": script_status("${VOICENOTES_MTIME}"),
            "lastModifiedDays": days_since("${VOICENOTES_MTIME}"),
            "commands": [
                "python3 ~/scripts/process-voice-notes.py",
                "tail -20 ~/scripts/voice-notes.log"
            ]
        },
        {
            "id": "clawdstrike",
            "name": "collect_verified.sh",
            "path": "/home/ubuntu/.openclaw/skills/clawdstrike/scripts/collect_verified.sh",
            "status": script_status("${CLAWDSTRIKE_MTIME}"),
            "lastModifiedDays": days_since("${CLAWDSTRIKE_MTIME}"),
            "commands": [
                "bash ~/.openclaw/skills/clawdstrike/scripts/collect_verified.sh"
            ]
        }
    ]
}
print(json.dumps(data, indent=2))
PYEOF
) > "$OUT"
```
