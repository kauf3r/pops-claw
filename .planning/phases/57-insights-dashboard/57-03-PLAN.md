    **Also create `src/components/hub/growth-sparkline.tsx`** -- a parameterized sparkline that accepts a chartColor prop (per anti-pattern: don't copy-paste OuraSparkline 4 times):

    ```typescript
    "use client";

    import { AreaChart, Area, ResponsiveContainer } from "recharts";

    interface GrowthSparklineProps {
      label: string;
      data: Array<{ v: number }>;
      chartColor?: string; // CSS variable like "var(--chart-3)"
    }

    export function GrowthSparkline({ label, data, chartColor = "var(--primary)" }: GrowthSparklineProps) {
      if (data.length === 0) return <div className="h-10 w-full" />;

      const color = `hsl(${chartColor})`;

      return (
        <div className="h-10 w-full">
          <ResponsiveContainer width="100%" height="100%">
            <AreaChart data={data} margin={{ top: 2, right: 0, left: 0, bottom: 0 }}>
              <defs>
                <linearGradient id={`spark-${label}`} x1="0" y1="0" x2="0" y2="1">
                  <stop offset="5%" stopColor={color} stopOpacity={0.3} />
                  <stop offset="95%" stopColor={color} stopOpacity={0} />
                </linearGradient>
              </defs>
              <Area
                type="monotone"
                dataKey="v"
                stroke={color}
                strokeWidth={1.5}
                fill={`url(#spark-${label})`}
                dot={false}
                isAnimationActive={false}
                connectNulls
              />
            </AreaChart>
          </ResponsiveContainer>
        </div>
      );
    }
    ```

    ```typescript
      if (!insights.weekStart) {

            <CardHeader className="pb-2">
              <CardTitle className="text-sm font-normal flex items-center gap-2 text-muted-foreground">

              </p>
            </CardContent>
          </Card>
        );
      }

      // Format weekStart date for display
      const reviewDate = new Date(insights.weekStart + "T12:00:00.000Z");
      const formattedDate = reviewDate.toLocaleDateString("en-US", {
        month: "long",
        day: "numeric",
      });

      return (
        <Card className="col-span-full">
          <CardHeader className="pb-2">
            <CardTitle className="text-sm font-normal flex items-center gap-2 text-muted-foreground">
              <Lightbulb className="h-4 w-4" />

            <p className="text-xs text-muted-foreground">

            </p>
          </CardHeader>
          <CardContent className="space-y-4">
            {insights.correlations.length > 0 && (

                <p className="text-xs font-bold uppercase tracking-widest text-muted-foreground/60">

              </div>
            )}
            {insights.themes.length > 0 && (
              <div className="space-y-2">
                <p className="text-xs font-bold uppercase tracking-widest text-muted-foreground/60">
                  Journal Themes

        </Card>
      );
    ```
