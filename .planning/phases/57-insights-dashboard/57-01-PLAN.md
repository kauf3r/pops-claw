```typescript
async function resolveUserId(): Promise<string | null> {
  const session = await auth.api.getSession({ headers: await headers() });
  if (session) return session.user.id;
  const headersList = await headers();
  const authHeader = headersList.get("authorization");
  if (!authHeader?.startsWith("Bearer ")) return null;
  const token = authHeader.slice(7);
  const expectedKey = process.env.GROWTH_API_KEY;
  if (!expectedKey || token !== expectedKey) return null;
  return process.env.GROWTH_DEFAULT_USER_ID ?? null;
}
```

    ```typescript
    import { eq, and, desc, sql } from "drizzle-orm";
    import { db } from "@/lib/db";
    import { habit, habitLog, ouraSnapshot, weeklyReview } from "@/lib/schema";

    // Types
    export interface HabitsSummaryResult {
      activeCount: number;
      bestStreak: { days: number; habitName: string } | null;
      todayCompleted: number;
      todayTotal: number;
    }

    export interface OuraGrowthResult {
      avgReadiness: number | null;
      avgSleep: number | null;
      sparklineData: Array<{ v: number }>;
    }

    export interface WeeklyInsightsResult {
      weekStart: string | null;
      correlations: Array<{ text: string }>;
      themes: Array<{ text: string; category: string; count: number }>;
    }

    // 1. getHabitsSummary
    export async function getHabitsSummary(userId: string): Promise<HabitsSummaryResult> {
      // Query active habits
      const habits = await db.select({
        name: habit.name,
        streakCurrent: habit.streakCurrent,
        streakBest: habit.streakBest,
        sourceId: habit.sourceId,
      }).from(habit)
        .where(and(eq(habit.userId, userId), eq(habit.status, "active")));

      if (habits.length === 0) return { activeCount: 0, bestStreak: null, todayCompleted: 0, todayTotal: 0 };

      // Find best current streak
      const best = habits.reduce((a, b) => ((a.streakCurrent ?? 0) >= (b.streakCurrent ?? 0) ? a : b));

      // Count today's completions
      const today = new Date().toISOString().split("T")[0]!;
      const todayLogs = await db.select({ habitSourceId: habitLog.habitSourceId })
        .from(habitLog)
        .where(and(
          eq(habitLog.userId, userId),
          eq(habitLog.date, today),
          eq(habitLog.status, "done")
        ));

      const activeSourceIds = new Set(habits.map(h => h.sourceId));
      const completedToday = todayLogs.filter(l => activeSourceIds.has(l.habitSourceId)).length;

      return {
        activeCount: habits.length,
        bestStreak: { days: best.streakCurrent ?? 0, habitName: best.name },
        todayCompleted: completedToday,
        todayTotal: habits.length,
      };
    }

    // 2. getOuraGrowthData -- 7-day avg readiness + sleep sparkline
    export async function getOuraGrowthData(userId: string): Promise<OuraGrowthResult> {
      const sevenDaysAgo = new Date();
      sevenDaysAgo.setDate(sevenDaysAgo.getDate() - 7);
      const startDate = sevenDaysAgo.toISOString().split("T")[0]!;

      const snapshots = await db.select({
        date: ouraSnapshot.date,
        sleepScore: ouraSnapshot.sleepScore,
        readinessScore: ouraSnapshot.readinessScore,
      }).from(ouraSnapshot)
        .where(and(eq(ouraSnapshot.userId, userId), sql`${ouraSnapshot.date} >= ${startDate}`))
        .orderBy(ouraSnapshot.date);

      if (snapshots.length === 0) return { avgReadiness: null, avgSleep: null, sparklineData: [] };

      const readinessScores = snapshots.map(s => s.readinessScore).filter((s): s is number => s != null);
      const sleepScores = snapshots.map(s => s.sleepScore).filter((s): s is number => s != null);

      return {
        avgReadiness: readinessScores.length > 0 ? Math.round(readinessScores.reduce((a,b) => a+b, 0) / readinessScores.length) : null,
        avgSleep: sleepScores.length > 0 ? Math.round(sleepScores.reduce((a,b) => a+b, 0) / sleepScores.length) : null,
        sparklineData: snapshots.map(s => ({ v: s.sleepScore ?? 0 })),
      };
    }

    // 3. getWeeklyInsights -- latest weekly review correlations + themes
    export async function getWeeklyInsights(userId: string): Promise<WeeklyInsightsResult> {
      const [latest] = await db.select({
        weekStart: weeklyReview.weekStart,
        correlationsJson: weeklyReview.correlationsJson,
        themesJson: weeklyReview.themesJson,
      }).from(weeklyReview)
        .where(eq(weeklyReview.userId, userId))
        .orderBy(desc(weeklyReview.createdAt))
        .limit(1);

      if (!latest) return { weekStart: null, correlations: [], themes: [] };

      let correlations: Array<{ text: string }> = [];
      let themes: Array<{ text: string; category: string; count: number }> = [];

      try { correlations = JSON.parse(latest.correlationsJson ?? "[]"); } catch {}
      try { themes = JSON.parse(latest.themesJson ?? "[]"); } catch {}

      return { weekStart: latest.weekStart, correlations, themes };
    }
    ```
