```typescript
// Growth Module Tables (Goals & Journal)

export const goal = pgTable("goal", {
  id: uuid("id").primaryKey().default(sql`gen_random_uuid()`),
  userId: text("user_id").notNull().references(() => user.id, { onDelete: "cascade" }),
  title: text("title").notNull(),
  description: text("description"), // plain text description of the objective
  keyResults: jsonb("key_results").$type<{ text: string; targetPct: number; currentPct: number }[]>().default([]),
  category: text("category"), // health, career, learning, financial, personal
  status: text("status").default("active").notNull(), // active, completed, abandoned
  targetDate: timestamp("target_date"),
  progress: real("progress").default(0).notNull(), // computed: average of KR currentPct values
  completedAt: timestamp("completed_at"),
  createdAt: timestamp("created_at").defaultNow().notNull(),
  updatedAt: timestamp("updated_at").defaultNow().$onUpdate(() => new Date()).notNull(),
}, (table) => [
  index("goal_user_idx").on(table.userId),
  index("goal_status_idx").on(table.status),
]);

export const goalCheckin = pgTable("goal_checkin", {
  id: uuid("id").primaryKey().default(sql`gen_random_uuid()`),
  goalId: uuid("goal_id").notNull().references(() => goal.id, { onDelete: "cascade" }),
  userId: text("user_id").notNull().references(() => user.id, { onDelete: "cascade" }),
  progressNotes: text("progress_notes"),
  blockers: text("blockers"),
  confidence: integer("confidence"), // 1-5 how confident in achieving the goal
  snapshotProgress: real("snapshot_progress"), // goal progress at time of check-in
  createdAt: timestamp("created_at").defaultNow().notNull(),
}, (table) => [
  index("goal_checkin_goal_idx").on(table.goalId),
  index("goal_checkin_user_idx").on(table.userId),
  index("goal_checkin_date_idx").on(table.createdAt),
]);

export const journalEntry = pgTable("journal_entry", {
  id: uuid("id").primaryKey().default(sql`gen_random_uuid()`),
  userId: text("user_id").notNull().references(() => user.id, { onDelete: "cascade" }),
  date: timestamp("date").notNull(), // one entry per user per day
  category: text("category").notNull(), // gratitude, challenges, learning, goals, wins, freeform, reflection
  prompt: text("prompt").notNull(),
  response: text("response"),
  mood: integer("mood"), // 1-5
  energy: integer("energy"), // 1-5
  tags: jsonb("tags").$type<string[]>().default([]),
  createdAt: timestamp("created_at").defaultNow().notNull(),
  updatedAt: timestamp("updated_at").defaultNow().$onUpdate(() => new Date()).notNull(),
}, (table) => [
  index("journal_entry_user_idx").on(table.userId),
  index("journal_entry_date_idx").on(table.date),
  uniqueIndex("journal_entry_user_date_idx").on(table.userId, table.date),
]);
```

Hash function: simple string hash (same approach as date-based deterministic selection):
```typescript
function simpleHash(str: string): number {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash |= 0;
  }
  return Math.abs(hash);
}
```
