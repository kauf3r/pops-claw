        payload = c.get('payload', c.get('text', ''))

        has_insights = 'key_insights' in payload
        has_response = 'response_text' in payload
        print(f'Cron: {c[\"name\"]}')

        print(f'  response linking: {\"YES\" if has_response else \"MISSING\"}')
