SSH to EC2 and run SQL against ~/clawd/content.db:

1. Create topic_suggestions table (user DMs Bob a subject, Bob inserts, Vector expands into topic briefs):
```sql
CREATE TABLE IF NOT EXISTS topic_suggestions (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  subject TEXT NOT NULL,
  source TEXT DEFAULT 'user',
  notes TEXT,
  status TEXT DEFAULT 'pending' CHECK(status IN ('pending', 'processing', 'expanded', 'rejected')),
  expanded_count INTEGER DEFAULT 0,
  created_at TEXT DEFAULT CURRENT_TIMESTAMP,
  updated_at TEXT DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX IF NOT EXISTS idx_suggestions_status ON topic_suggestions(status);
```

2. Create content_pillars table (4-5 pillar topics for SEO architecture):
```sql
CREATE TABLE IF NOT EXISTS content_pillars (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL UNIQUE,
  slug TEXT NOT NULL UNIQUE,
  description TEXT,
  target_cluster_count INTEGER DEFAULT 10,
  status TEXT DEFAULT 'active' CHECK(status IN ('active', 'archived')),
  created_at TEXT DEFAULT CURRENT_TIMESTAMP,
  updated_at TEXT DEFAULT CURRENT_TIMESTAMP
);
```

3. Create content_clusters table (8-15 cluster articles per pillar):
```sql
CREATE TABLE IF NOT EXISTS content_clusters (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  pillar_id INTEGER NOT NULL REFERENCES content_pillars(id),
  name TEXT NOT NULL,
  slug TEXT NOT NULL,
  description TEXT,
  target_article_count INTEGER DEFAULT 3,
  status TEXT DEFAULT 'active' CHECK(status IN ('active', 'saturated', 'archived')),
  created_at TEXT DEFAULT CURRENT_TIMESTAMP,
  updated_at TEXT DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX IF NOT EXISTS idx_clusters_pillar ON content_clusters(pillar_id);
CREATE INDEX IF NOT EXISTS idx_clusters_status ON content_clusters(status);
```
